package chaper1;

public class gugudan {

	public static void main(String[] args) {
		//2~9단까지의 구구단을 출력하는 프로그램 while 문 쓰기
		int i = 2;
	while(i < 10) {
		System.out.println(i + "단을 출력합니다");
		int j = 1;
		while(j < 10) {
			System.out.println(i + "x" + j + " = " + (i * j));
			j++;
		}
		System.out.println();
		i++;
	}
	}

}
