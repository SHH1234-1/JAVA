package chaper1;

public class gugudan {

	public static void main(String[] args) {
		//2~9단까지의 구구단을 출력하는 프로그램
		
		
		for(int i = 2; i < 10; i++) {
			System.out.println(i + "단의 구구단을 출력합니다");
			for(int j = 1;j < 10; j++) {
				System.out.println(i + "x" + j + "=" + (i * j));
			}
			System.out.println("");
		}

	}

}
